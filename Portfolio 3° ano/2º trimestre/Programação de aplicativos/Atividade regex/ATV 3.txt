function verificarTipoCaracter(caracter) {
  var regex = /^[A-Za-z!@#$%^&*()_+~`|}{\[\]:;"'<,>.\/?-]+$/;
  return regex.test(caracter);
}

var caracterDigitado = "A";
var tipoCaracterPermitido = verificarTipoCaracter(caracterDigitado);
